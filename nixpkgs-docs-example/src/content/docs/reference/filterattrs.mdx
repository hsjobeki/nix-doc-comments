Filter an attribute set by removing all attributes for which the
given predicate return false.

# Example

filterAttrs (n: v: n == "foo") { foo = 1; bar = 2; }
=> { foo = 1; }

# Type

filterAttrs :: (String -> Any -> Bool) -> AttrSet -> AttrSet
