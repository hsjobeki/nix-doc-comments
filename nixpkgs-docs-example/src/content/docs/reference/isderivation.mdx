Check whether the argument is a derivation. Any set with
`{ type = "derivation"; }` counts as a derivation.

# Example

nixpkgs = import <nixpkgs> {}
isDerivation nixpkgs.ruby
=> true
isDerivation "foobar"
=> false

# Type

isDerivation :: Any -> Bool
