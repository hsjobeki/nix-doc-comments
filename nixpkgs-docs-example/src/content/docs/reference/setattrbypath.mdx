import { LinkCard, CardGrid } from '@astrojs/starlight/components';

Create a new attribute set with `value` set at the nested attribute location specified in `attrPath`.

# Example

setAttrByPath ["a" "b"] 3
=> { a = { b = 3; }; }

# Type

setAttrByPath :: [String] -> Any -> AttrSet
