If `cond` is true, return the attribute set `as`,
otherwise an empty attribute set.

# Example

optionalAttrs (true) { my = "set"; }
=> { my = "set"; }
optionalAttrs (false) { my = "set"; }
=> { }

# Type

optionalAttrs :: Bool -> AttrSet -> AttrSet
